
#include<bits/stdc++.h>
using namespace std;
#define ll long long
vector<vector<ll>>adj(10000);
vector<bool>vis(10000,false);

void bfs(int n)
{
    queue<ll>q;
    q.push(n);
    while(q.empty()==false)
    {
        ll so = q.front();
        q.pop();
        if(vis[so]) continue;
        vis[so]=true;
        cout << so << ' ';  
        for(auto i : adj[so])
        {
            if(vis[i]==false) q.push(i);
        }
    }
}
int main()
{
    ll n,e;
    cin >> n >> e;
    for(int i=0 ; i<e ; i++) ///O(e)
    {
        ll x,y;
        cin >> x >> y;
        adj[x].push_back(y);
        adj[y].push_back(x);
    }
    bfs(1); 
    
}
