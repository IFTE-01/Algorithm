#include<bits/stdc++.h>
using namespace std;
#define ll long long

vector<vector<pair<ll,ll>>>adj(100000);
vector<bool>vis(100000,false);
ll prims(int n)
{
    priority_queue<pair<ll,ll>,vector<pair<ll,ll>>,greater<pair<ll,ll>>>pq;
    pq.push({0,n});
    ll total = 0;
    while(pq.empty()==false)
    {
        ll so = pq.top().second;
        ll dis = pq.top().first;
        pq.pop();
        if(vis[so]) continue;
        vis[so]=true;
        total+=dis;
        for(auto c : adj[so])
        {
            if(vis[c.second]==false)
            {
                pq.push({c.first,c.second});
            }
        }
    }
    return total;
}

int main()
{
    ll n,e;
    cin >> n >> e;
    for(int i=0 ; i<e ; i++)
    {
        ll x,y,w;
        cin >> x >> y >> w;
        adj[x].push_back({w,y});
        adj[y].push_back({w,x});
    }

    ll node;
    cin >> node;
    cout << prims(node);
}

input:
5 6
0 1 2
0 2 4
1 2 1
1 3 7
2 4 3
3 4 1
0

output:
7
