#include<bits/stdc++.h>
using namespace std;
#define ll long long
vector<vector<ll>>adj(10000);
vector<bool>vis(10000,false);

void dfs(int n)
{
    vis[n]=true;
    cout << n << ' ';
    for(auto i : adj[n])
    {
        if(vis[i]==false) dfs(i);
    }
}
int main()
{
    ll n,e;
    cin >> n >> e;
    for(int i=0 ; i<e ; i++)  
    {
        ll x,y;
        cin >> x >> y;
        adj[x].push_back(y);
        adj[y].push_back(x);
    }
    dfs(1);  
}
